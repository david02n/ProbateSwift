CREATE TABLE applicants (
  id                SERIAL PRIMARY KEY,
  case_id           INT         NOT NULL REFERENCES probate_cases(id),
  user_id           INT         NOT NULL REFERENCES users(id),

  person_position   SMALLINT    NOT NULL CHECK (person_position BETWEEN 1 AND 4),
  executor          BOOLEAN     NOT NULL DEFAULT FALSE,
  title             VARCHAR(10),
  first_names       VARCHAR(255) NOT NULL,
  middle_names      VARCHAR(255),
  last_name         VARCHAR(255) NOT NULL,
  name_diff_will    BOOLEAN     NOT NULL DEFAULT FALSE,
  alt_name_will     VARCHAR(255),
  address_line1     VARCHAR(255) NOT NULL,
  address_line2     VARCHAR(255),
  city              VARCHAR(100) NOT NULL,
  county            VARCHAR(100),
  postcode          VARCHAR(20)  NOT NULL,
  phone_home        VARCHAR(50),
  phone_mobile      VARCHAR(50),
  email             VARCHAR(255),
  relationship      VARCHAR(100),
  is_notifying      BOOLEAN     NOT NULL DEFAULT FALSE,
  status            VARCHAR(50) NOT NULL DEFAULT 'pending',
  created_at        TIMESTAMP WITHOUT TIME ZONE NOT NULL DEFAULT now(),
  updated_at        TIMESTAMP WITHOUT TIME ZONE NOT NULL DEFAULT now(),

  CONSTRAINT uq_case_position UNIQUE(case_id, person_position),
  CONSTRAINT chk_executor_fields
    CHECK (
      (executor = TRUE  AND name_diff_will IN (TRUE, FALSE))
      OR (executor = FALSE AND name_diff_will = FALSE AND alt_name_will IS NULL)
    )
);
